## 一、目标概述

根据 `MEMORY_UPLOAD.md` 设计文档，重构 `chat_init` 流程，实现多层记忆架构：
1. **角色核心锚点** (100-200字) - 始终加载
2. **用户画像** (结构化JSON) - 始终加载
3. **记忆库** (追加存储) - RAG检索
4. **原始记忆备份** - 用于重新生成

**约束条件**：
- 先用 DeepSeek 测试，MiniMax 配置预留
- 保持 MQ 调用方式不变

## 三、实现细节

### 3.1 chat_init 完整流程

```python
async def chat_init(body):
    msg_json = json.loads(body)
    data_json = msg_json.get("data")
    
    user_id = data_json.get("userId")
    experience = data_json.get("experience")
    llm_id = data_json.get("llmId")
    
    # Step 1: 存入原始记忆备份
    raw_key = build_key(LLMChatConstant.RAW_EXPERIENCE, user_id, llm_id)
    redis_client.set(raw_key, experience)
    
    # Step 2: LLM提炼角色核心锚点
    core_anchor = await _extract_core_anchor(experience)
    
    # Step 3: LLM生成用户画像
    user_profile = await _generate_user_profile(experience)
    
    # Step 4: LLM提取初始事件
    initial_events = await _extract_initial_events(experience)
    
    # Step 5: 分离存储
    core_key = build_key(LLMChatConstant.CORE_ANCHOR, user_id, llm_id)
    profile_key = build_key(LLMChatConstant.USER_PROFILE, user_id, llm_id)
    bank_key = build_key(LLMChatConstant.MEMORY_BANK, user_id, llm_id)
    
    redis_client.set(core_key, core_anchor)
    redis_client.set(profile_key, json.dumps(user_profile))
    redis_client.set(bank_key, json.dumps(initial_events))
    
    # 兼容旧逻辑：同时存入 init_memory (便于后续迁移)
    init_key = LLMChatConstant.CHAT_MEMORY + user_id + ":" + llm_id + ":" + LLMChatConstant.INIT_MEMORY
    redis_client.set(init_key, experience)
```

### 8.3 Step 3: 重构 memory_upload_service.py

**文件**: `app/service/memory_upload_service.py`

**完整替换为**:
```python
import json
from loguru import logger

from app.common.constant.LLMChatConstant import LLMChatConstant
from app.core import redis_client
from app.core.llm_model.model import LLM_MAP
from app.core.prompts.prompt_manager import PromptManager


def build_key(prefix: str, user_id: str, llm_id: str) -> str:
    return f"{prefix}{user_id}:{llm_id}"


async def _call_llm(prompt_template: str, variables: dict, model_name: str = "ds_model") -> str:
    prompt = PromptManager.get_prompt(prompt_template)
    llm = LLM_MAP.get(model_name)
    chain = prompt | llm
    return await chain.ainvoke(variables)


async def _extract_core_anchor(experience: str) -> str:
    return await _call_llm("role_core_anchor.md", {"raw_experience": experience})


async def _generate_user_profile(experience: str) -> dict:
    result = await _call_llm("user_profile.md", {"raw_experience": experience}, "ds_model")
    return json.loads(result)


async def _extract_initial_events(experience: str) -> list:
    result = await _call_llm("memory_event_extractor.md", {"raw_experience": experience}, "ds_model")
    return json.loads(result)


async def chat_init(body):
    """
    模型经历上传 - 多层记忆架构
    """
    msg_json = json.loads(body)
    data_json = msg_json.get("data")
    
    user_id = data_json.get("userId")
    experience = data_json.get("experience")
    llm_id = data_json.get("llmId")
    
    logger.info(f"开始处理用户 {user_id} 的初始记忆...")
    
    raw_key = build_key(LLMChatConstant.RAW_EXPERIENCE, user_id, llm_id)
    redis_client.set(raw_key, experience)
    
    try:
        core_anchor = await _extract_core_anchor(experience)
        core_key = build_key(LLMChatConstant.CORE_ANCHOR, user_id, llm_id)
        redis_client.set(core_key, core_anchor)
    except Exception as e:
        logger.error(f"角色核心锚点生成失败: {e}")
        core_anchor = ""
    
    try:
        user_profile = await _generate_user_profile(experience)
        profile_key = build_key(LLMChatConstant.USER_PROFILE, user_id, llm_id)
        redis_client.set(profile_key, json.dumps(user_profile, ensure_ascii=False))
    except Exception as e:
        logger.error(f"用户画像生成失败: {e}")
        user_profile = {}
    
    try:
        initial_events = await _extract_initial_events(experience)
        bank_key = build_key(LLMChatConstant.MEMORY_BANK, user_id, llm_id)
        redis_client.set(bank_key, json.dumps(initial_events, ensure_ascii=False))
    except Exception as e:
        logger.error(f"初始事件提取失败: {e}")
        initial_events = []
    
    init_key = LLMChatConstant.CHAT_MEMORY + user_id + ":" + llm_id + ":" + LLMChatConstant.INIT_MEMORY
    redis_client.set(init_key, experience)
    
    logger.info(f"用户 {user_id} 的初始记忆处理完成")
```
